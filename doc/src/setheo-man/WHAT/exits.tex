\section{Return values of the SETHEO system}
\begin{description}
\item[0]
	print usage or print verbosity levels {\tt inwasm -verbose}
\item[0]
	no input file specified
\item[1]
	error in command-line parameters
\item[2]
	cannot open input or output files
\item[3]
	command parameters are not consistent
\item[1]
	syntax errors occurred (abort after parsing)
\item[4]
	purity removed all clauses. ``Fail'' is generated.
\item[1]
	errors occurred during preprocessing (before code
	generation)
\item[0]
	successful termination
\item[7]
	code generation: out of memory
\item[7]
	code generation: undefined labels during pass2
\item[11]
	code generation: symbol table overflow
\item[9]
	code generation: optimizer internal: must be a label

\end{description}
