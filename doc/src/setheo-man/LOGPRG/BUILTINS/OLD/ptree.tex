\subsection{\$ptree/0}

\begin{description}
\item[Synopsis:]
	{\tt \$ptree}
\item[Low Level Name:]
	{\tt ptree }
\item[Result:]\ \\
\end{description}

\vspace*{0.5cm}
\noindent
{\bf Description.}
When this built-in is encountered, it appends the current tableau
to the file {\em file}.tree.
