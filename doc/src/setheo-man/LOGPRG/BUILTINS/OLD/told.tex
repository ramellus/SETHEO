\subsection{\$told/0}

\begin{description}
\item[Synopsis:]
	{\tt \$told}
\item[Parameters:]\ \\[-0.5cm]
\item[Low Level Name:]
	{\tt told}
\item[Result:]\ \\
this built-in always succeeds
\end{description}

\vspace*{0.5cm}
\noindent
{\bf Description.}
When encountered, this built-in closes a named file which have previously
been opened by {\tt \$tell(T)}.

\vspace*{0.5cm}
\noindent
{\bf Side-effects.}

\vspace*{0.5cm}
\noindent
{\bf Notes.}

\vspace*{0.5cm}
\noindent
{\bf Example.}
\begin{verbatim}
p(X) <- $tell("output"),$write(X),$told.
\end{verbatim}
