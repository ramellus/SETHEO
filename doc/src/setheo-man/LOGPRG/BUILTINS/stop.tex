\subsection{\$stop/0}

\begin{description}
\item[Synopsis:]
	{\tt \$stop}
\item[Low Level Name:]
	{\tt stop}
\end{description}

\vspace*{0.5cm}
\noindent
{\bf Description.}
This built-in unconditionally stops the run of the \SAM\ with a success
message. Any illegal usage of this built-in destroys soundness of
SETHEO.

\vspace*{0.5cm}
\noindent
{\bf Example.}
\begin{verbatim}
p(X) <- $write("End of run"),$stop.
\end{verbatim}
